Signup Function

Auth.signup({
  username: 'songz',
  email: 'song@zheng.club',
  password: 'helpless',
  name: 'aouo oauu',
  age: 942,
  //any other key-value pairs...
}).then(user => console.log('user is', user))

Login Function

Auth.login({
  username: 'songz',
  password: 'helpless',
}).then(user => console.log('user is', user))

Session Function

Auth.getSession().then( user => {
  console.log('User is', user)
})

Logout Function

// Logout, then get session should be empty (jwt token cleared out of localstorage)
Auth.logout()
